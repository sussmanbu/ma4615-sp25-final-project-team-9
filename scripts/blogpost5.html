# Step 3: Clean data & compute total enrollment per school
enrollment_clean <- enrollment %>%
  mutate(across(all_of(student_columns), ~ ifelse(. < 0, 0, .))) %>%
  rowwise() %>%
  mutate(Total_Students = sum(c_across(all_of(student_columns)), na.rm = TRUE)) %>%
  ungroup()

  filter(LEA_STATE == "MA") %>%
