  filter(LEA_STATE_NAME == "MASSACHUSETTS")

  inner_join(harassment, by = "COMBOKEY") 
merged <- merged %>%
  mutate(
    total_allegations = rowSums(select(., SCH_HBALLEGATIONS_SEX, SCH_HBALLEGATIONS_RAC, SCH_HBALLEGATIONS_DIS), na.rm = TRUE)
  )
